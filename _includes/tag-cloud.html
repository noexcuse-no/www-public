{% comment %} Tag cloud — displays all article tags as clickable links with counts. Derives tags from all pages with class: article. Uses site.pages — works on any page within the site context. {% endcomment %} {% assign articles = site.pages | where: "class", "article" %} {% assign all_tags = "" | split: "," %} {% for p in articles %} {% if p.tags and p.tags.size > 0 %} {% assign all_tags = all_tags | concat: p.tags %} {% endif %} {% endfor %} {% assign unique_tags = all_tags | uniq | sort_natural %}
{% for tag in unique_tags %} {% assign count = 0 %} {% for p in articles %} {% if p.tags contains tag %} {% assign count = count | plus: 1 %} {% endif %} {% endfor %} {{ tag }} {{ count }} {% endfor %}
